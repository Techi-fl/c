Especificador    	Tipo de dado	    Exemplo de uso	                   Saída
%d ou %i	           (int)            printf("Idade: %d", 20);	       Idade: 20
%f            	      (float)	        printf("Nota: %.2f", 7.5);	       Nota: 7.50
%c	                  (char)         	printf("Letra: %c", 'A');	       Letra: A



#include <stdio.h>

int main(void) {
    int idade;
    
    printf("Digite sua idade: ");                        // Mostra a mensagem no terminal
    scanf("%d", &idade);                                // Lê o valor digitado e salva na variável

    printf("Você tem %d anos.\n", idade);              // Exibe o valor da variável
}




//while

#include <stdio.h>

int main(void) {
    int i = 1;

    while (i <= 5) {
        printf("Contando: %d\n", i);
        i++;  // incrementa
    }
}




//do... while

#include <stdio.h>

int main(void) {
    int i = 1;

    do {
        printf("Número: %d\n", i);
        i++;
    } while (i <= 3);
}



//for 
#include <stdio.h>

int main(void) {
    for (int i = 1; i <= 5; i++) {
        printf("Valor: %d\n", i);
    }
}



\\ if.. else
#include <stdio.h>

int main(void) {
    int idade = 18;

    if (idade >= 18) {
        printf("Você é maior de idade.\n");
    }
}


#include <stdio.h>

int main(void) {
    int nota;

    printf("Digite sua nota (0 a 10): ");
    scanf("%d", &nota);

    if (nota >= 9) {
        printf("Excelente!\n");
    } else if (nota >= 7) {
        printf("Boa nota.\n");
    } else if (nota >= 5) {
        printf("Na média.\n");
    } else {
        printf("Reprovado.\n");
    }
}
